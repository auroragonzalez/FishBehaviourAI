# Setup: Activate environment and download datasets
source cnr/bin/activate

# Download datasets from Roboflow
# Datasets will be downloaded to data/datasets/
python << 'EOF'
from roboflow import Roboflow
import os

# Change to datasets directory
os.chdir('data/datasets')

rf = Roboflow(api_key="YOUR_API_KEY")

# Video 1 - Main dataset with all behaviors
project = rf.workspace("fishbehaviour").project("dataset_video1")
version = project.version(1)
dataset = version.download("yolov9", location="./video1")
EOF
